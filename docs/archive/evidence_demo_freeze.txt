def _render_explorer_summary(evidence_items):
    platforms = {
        _format_value(_safe_get(item, "platform", "Unknown"))
        for item in evidence_items
    }

    topics = {
        _format_value(_safe_get(item, "topic", "Unknown"))
        for item in evidence_items
    }

    negative_count = len(
        [
            item
            for item in evidence_items
            if _format_value(_safe_get(item, "sentiment", "")).lower()
            in ["negative", "йһҺоәўоІ„"]

    col1.metric("Evidence", len(evidence_items))

    col3.metric("Topics", len(topics))

def _render_filters(evidence_items):
    st.markdown("### Filters")

    platforms = sorted(
        {
            _format_value(_safe_get(item, "platform", "Unknown"))
            for item in evidence_items
        }
    )

    topics = sorted(
        {
            _format_value(_safe_get(item, "topic", "Unknown"))
            for item in evidence_items
        }
    )

    sentiments = sorted(
        {
            _format_value(_safe_get(item, "sentiment", "Unknown"))
            for item in evidence_items
        }
    )

    col1, col2, col3, col4 = st.columns(4)

    with col1:
        platform_filter = st.selectbox(
            "Platform",
            options=["All"] + platforms,
            key="evidence_explorer_platform_filter",
        )

    with col2:
        topic_filter = st.selectbox(
            "Topic",
            options=["All"] + topics,
            key="evidence_explorer_topic_filter",
        )

    with col3:
        sentiment_filter = st.selectbox(
            "Sentiment",
            options=["All"] + sentiments,
            key="evidence_explorer_sentiment_filter",
        )

    with col4:
        keyword = st.text_input(
            "Keyword",
            value="",
            key="evidence_explorer_keyword_filter",
        )

    filtered_items = []

    for item in evidence_items:
        platform = _format_value(_safe_get(item, "platform", "Unknown"))
        topic = _format_value(_safe_get(item, "topic", "Unknown"))
        sentiment = _format_value(_safe_get(item, "sentiment", "Unknown"))

        title = _format_value(_safe_get(item, "title", ""))
        content = _format_value(_safe_get(item, "content", ""))

        if platform_filter != "All" and platform != platform_filter:
            continue

        if topic_filter != "All" and topic != topic_filter:
            continue

        if sentiment_filter != "All" and sentiment != sentiment_filter:
            continue

        if keyword:
            search_text = f"{title} {content}".lower()

            if keyword.lower() not in search_text:
                continue

        filtered_items.append(item)

    st.caption(f"?жЎҖ?жҶҝиҲҗе…§ {len(filtered_items)} / {len(evidence_items)} ??Evidence")

    return filtered_items

        evidence_items,

def _find_evidence(evidence_items, evidence_id):
    if evidence_id is None:
        return None

    evidence_id = str(evidence_id)

    for item in evidence_items:
        if _get_evidence_id(item) == evidence_id:
            return item

    return None


def _get_evidence_id(item):
    evidence_id = _safe_get(item, "evidence_id", None)

    if evidence_id is None:
        evidence_id = _safe_get(item, "id", None)

    if evidence_id is None:
        evidence_id = _safe_get(item, "title", "unknown_evidence")

    return str(evidence_id)


def _safe_get(item, key, default=None):
    if isinstance(item, dict):
        return item.get(key, default)

    return getattr(item, key, default)


def _safe_number(value):
    try:
        return int(value)
    except (TypeError, ValueError):
        return 0


def _format_value(value):
    if value is None:
        return "Unknown"

    if hasattr(value, "value"):
        return str(value.value)

    return str(value)


def _to_dict(item):
    if isinstance(item, dict):
        return item

    result = {}

    for key in dir(item):
        if key.startswith("_"):
            continue

        value = getattr(item, key)

        if callable(value):
            continue

        result[key] = _format_value(value)

    return result

def render_hub_runtime_context_panel(runtime_context: dict):

    st.markdown("## Runtime Context")

    if not runtime_context:
        st.warning("No runtime context available.")
        return

    for layer_name, layer_state in runtime_context.items():
        connected = layer_state.get("connected") is True
        status_label = "Connected" if connected else "Not Connected"

        with st.container(border=True):
            col1, col2 = st.columns([3, 1])

            with col1:
                st.markdown(f"### {layer_name}")
                st.caption(status_label)

                runtime_type = layer_state.get("runtime_type")
                if runtime_type:
                    st.write(f"Runtime Type: `{runtime_type}`")

            with col2:
                st.metric("Status", status_label)

            available_methods = layer_state.get("available_methods", [])
            missing_methods = layer_state.get("missing_methods", [])

            method_col1, method_col2 = st.columns(2)

            with method_col1:
                st.markdown("#### Available Methods")

def _render_list(items, empty_text="No items available."):
    if not items:
        st.caption(empty_text)
        return

    for item in items:
        st.markdown(f"- {item}")
